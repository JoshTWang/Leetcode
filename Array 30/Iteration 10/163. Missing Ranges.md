class Solution {
    public List<String> findMissingRanges(int[] nums, int lower, int upper) {
        List<String> res = new ArrayList<>();
        int start = lower;
        int end;
        for (int i = 0; i <= nums.length; i += 1) {
            if (i == nums.length) {
                // last check -- nums[nums.length - 1] -- upper
                end = upper;
            } else {
                end = nums[i] - 1;
            }
            
            if (start == end) {
                res.add("" + start);
            } else if (start < end) {
                res.add(start + "->" + end);
            }            
            start = end + 2;
        }
        return res;
    }
}
