package _69_Sqrt;

class MySolution {
    public int mySqrt(int x) {
        if (x < 2) {
            return x;
        }
        long num;
        int left = 0;
        int right = x / 2;
        int mid;
        while (left <= right) {
            mid = left + (right - left) / 2;
            num = (long)mid * mid; // long variable in case the input is too large
            if (num > x) {
                right = mid - 1;
            } else if (num < x) {
                left = mid + 1;
            } else {
                return mid;
            }
        }
        return right; // because when sqrt(8) = 2 -- We want the lower bound
    }
}
